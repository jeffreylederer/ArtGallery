using System.Text.RegularExpressions;
using System.Text;
using System.Data;




	public static Diagnosis.MasterDiagnosisTable Get(string searchString)
        {
            Regex rex = new Regex( "[A-Za-z0-9]+" );
            MatchCollection mc = rex.Matches( searchString );
            if (mc.Count == 0)
                return new MasterDiagnosisTable();
            StringBuilder query = new StringBuilder();
            query.Append( "select distinct id, code, description " );
            query.Append( "from Picture where "();
            for (int i = 0; i < mc.Count; i++)
            {
                if (i == mc.Count - 1)
                    query.AppendFormat(Contains(description, '{0}')", mc[i].Value ))
                else
                    query.AppendFormat( "Contains(description, '{0}') and", mc[i].Value );
            }
	    query.Append(")");
	    
	    // do something with code using like, we can talk about this

            string commandtext = query.ToString();
            SqlConnection conn = new SqlConnection( ConfigurationManager.ConnectionStrings["GalleryConnectionString"].ConnectionString );
            Diagnosis data = new Diagnosis();
            try
            {
                conn.Open();
                SqlCommand selectCommand = new SqlCommand( commandtext, conn );
                selectCommand.CommandType = CommandType.Text;
                SqlDataAdapter da = new SqlDataAdapter( selectCommand );
                da.Fill( data, "MasterDiagnosis");
            }
            catch(Exception ex)
            {
                string message = ex.Message;
            }
            finally
            {
                if (conn != null && conn.State == ConnectionState.Open)
                    conn.Close();
            }
            return data.MasterDiagnosis;
        }
